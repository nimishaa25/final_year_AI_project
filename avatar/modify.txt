add this in main.py

from fastapi import FastAPI
from avatar.video_pipeline import generate_avatar_video

app = FastAPI()

@app.post("/generate-avatar")
async def generate_avatar(request: dict):
    text = request.get("text")

    if not text:
        return {"error": "No text provided"}

    result = generate_avatar_video(text)

    return {"message": "Video Generated", "path": result}
